using Leopotam.Ecs;
using UnityEngine;

namespace #NS# {
    sealed class #SCRIPTNAME# : MonoBehaviour {
        EcsWorld _world;
        EcsSystems _systems;

        void OnEnable () {
            _world = new EcsWorld ();
            _systems = new EcsSystems (_world);
#if UNITY_EDITOR
            Leopotam.Ecs.UnityIntegration.EcsWorldObserver.Create (_world);
            Leopotam.Ecs.UnityIntegration.EcsSystemsObserver.Create (_systems);
#endif
            _systems
                // Register your systems here, for example:
                // .Add (new TestSystem1 ())
                // .Add (new TestSystem2 ())
                .Init ();
        }

        void Update () {
            _systems.Run ();
            _world.EndFrame ();
        }

        void OnDisable () {
            _systems.Destroy ();
            _systems = null;
            _world.Destroy ();
            _world = null;
        }
    }
}
